Example: Submit a Job

curl -X POST http://localhost:8000/api/jobs \
  -H "Content-Type: application/json" \
  -d '{
    "asset_id": "your-asset-uuid",
    "template": "studio_luxury",
    "prompt": "展示高端连衣裙",
    "persona": "east_asian",
    "route": "standard",
    "duration_seconds": 8
  }'

Example: Enhance a Prompt

curl -X POST http://localhost:8000/api/prompts/enhance \
  -H "Content-Type: application/json" \
  -d '{
    "prompt": "夏季连衣裙",
    "template": "outdoor_cinematic",
    "persona": "luxury_editorial",
    "language": "zh-CN"
  }'
